\documentclass{article}
\usepackage{lisp-on-tex}
\lispinterp{%
  (\define \sq (\lambda (\n) (\* \n \n)))
  (\define \fact (\lambda (\n) (\lispif (\= \n :0) :1 (\* \n (\fact (\- \n :1))))))
  }
\newcommand\sq[1]{\lispinterp{(\texprint (\sq :#1))}}
\newcommand\fact[1]{\lispinterp{(\texprint (\fact :#1))}}
