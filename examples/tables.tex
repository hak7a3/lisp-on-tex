\documentclass[pdftex]{article}
\usepackage{lisp-on-tex}
\usepackage{graphicx}
\usepackage{color}
\lispinterp{%
  (\define \numlist (\list :"0 :"33 :"66 :"99 :"CC :"FF))
  (\define \getColor (\lambda (\n)
      (\list 
        (\nth \numlist (\/ \n :36)) 
        (\nth \numlist (\/ (\mod \n :36) :6)) 
        (\nth \numlist (\mod \n :6)))))
  %(\define)
}
\begin{document}
  \lispinterp{(\print (\getColor :5))}
\end{document}
