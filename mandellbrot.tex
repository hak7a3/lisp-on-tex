\documentclass{article}
\usepackage[dvipdfm,a4paper,margin=1pt]{geometry}
\def\w{\kern1pt}
\def\b{\vrule width 1pt height 1pt depth 0pt}
\usepackage{lisp}
\lispinterpl{%
(\define \maxloop :100)
(\define \scale :100)
(\define \limitr (\* :4 \scale \scale))
(\define \sq (\lambda (\x) (\/ (\* \x \x) \scale)))
(\define \isMandell 
   (\lambda (\a \b \k \x \y)
     (\lispif (\< \maxloop \k)
        /t
        (\lispif (\< \limitr (\+ (\sq \x) (\sq \y)))
           /f
           (\isMandell \a \b (\+ \k :1)
                      (\+ \a (\sq \x) (\- (\sq \y))) 
                      (\+ \b (\/ (\* :2 \x \y) \scale)))))))
(\define \drawMandell 
  (\lambda (\a \b) 
    (\lispif (\isMandell \a \b :0 :0 :0)
      (\begin
        (\texprint '\b')
        (\immediatewrite)
      ) 
      (\begin
        (\texprint '\w')
        (\immediatewrite)
      ))))
(\define \modulo (\lambda (\n \m) (\- \n (\* \m (\/ \n \m)))))
(\define \loopMandell
  (\lambda (\n)
    (\lispif (\= \n (\* :6 \scale \scale)) 
      ()
      (\begin
         (\drawMandell 
            (\- (\modulo \n (\* :3 \scale)) (\* :2 \scale)) 
            (\- (\/ \n (\* :3 \scale)) \scale))
         (\lispif (\= (\modulo \n (\* :3 \scale)) (\- (\* :3 \scale) :1))
             (\texprint '\\') ())
         (\loopMandell (\+ \n :1))))))
}

\lispinterpl{(\loopMandell :0)} 
\end{document}
